    <label>Altura (cm):</label>
    <input type="number" id="altura" placeholder="Ex: 30">
    
    <label>Preço por m² (R$):</label>
    <input type="number" id="preco" placeholder="Ex: 50" value="50">
    
    <button onclick="calcular()">Calcular</button>
    
    <div class="resultado" id="resultado"></div>
  </div>

  <script>
    function calcular() {
      let largura = document.getElementById("largura").value;
      let altura = document.getElementById("altura").value;
      let preco = document.getElementById("preco").value;

      if(largura > 0 && altura > 0 && preco > 0){
        let area = (largura/100) * (altura/100);
        let valor = area * preco;
        document.getElementById("resultado").innerHTML = 
          "Preço: R$ " + valor.toFixed(2);
      } else {
        document.getElementById("resultado").innerHTML = 
          "Preencha todos os campos corretamente.";
      }
    }
  </script>
